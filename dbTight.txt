// Use DBML to define your database structure

Table locations {
  id integer [primary key]
  company varchar
  city varchar
  country varchar
  website varchar
}

Table contacts {
  id integer [primary key]
  first_name varchar
  last_name varchar
  location_id integer [ref: > locations.id]
  notes varchar
}

Table machines {
  id integer [primary key]
  location_id integer [ref: > locations.id]
  owner_id integer [ref: > contacts.id]
  contact_id integer [ref: > contacts.id]
  type varchar
  model varchar
  year integer
  hours integer
  current_price varchar
  condition varchar
  acquiredDate datetime
  images array
  description varchar
  description_documents array
  inspection_documents array
  notes varchar
}

Table transactions {
  id integer [primary key]
  seller_company_id integer [ref: > locations.id]
  seller_contact_id integer [ref: > contacts.id]
  buyer_company_id integer [ref: > locations.id]
  buyer_contact_id integer [ref: > contacts.id]
  transportation_company_id integer [ref: > locations.id]
  salePrice varchar
  amountPaid varchar
  profitFromSale varchar
  notes varchar
}

Table archives {
  id integer [primary key]
  machine_id integer [ref: - machines.id]
  dateArchived datetime
}
